排序算法

2020-06-28 2 min read

0.排序模板

package com.lsz.数据结构与算法.sort;
import java.util.Arrays;
import java.util.concurrent.ThreadLocalRandom;

/**
 * 排序
 */
public class Sort {
    public static void main(String[] args) {
        Sort sort = new Sort();
        int[] array = sort.generateRandomArray(10, 30, 65);
        System.out.println("排序前:" + Arrays.toString(array));
        sort.selectionSort(array);
        System.out.println("排序后:" + Arrays.toString(array));
    }

    /**
     * 交换array中的下标为i,j的元素
     *
     * @param array
     * @param i
     * @param j
     */
    private  void swap(int[] array, int i, int j) {
        //这种交换不好，i=j时会犯错
//        array[i] = array[i] + array[j];
//        array[j] = array[i] - array[j];
//        array[i] = array[i] - array[j];
        int temp  =array[i];
        array[i] = array[j];
        array[j] = temp;
    }

    /**
     * 生成随机数组
     *
     * @param n
     * @param min
     * @param max
     * @return
     */
    public int[] generateRandomArray(int n, int min, int max) {
        int[] array = new int[n];
        for (int i = 0; i < n; i++) {
            array[i] = ThreadLocalRandom.current().nextInt(min, max + 1);
        }
        return array;
    }
}

1. 选择排序

遍历数组，把当前元素和后面未排序元素中最小的元素交换。

private int[] selectionSort(int[] array) {
        for (int i = 0; i < array.length; i++) {
            //找出[i,n]中最小的
            int minIndex = i;
            for (int j = i + 1; j < array.length; j++)
                if (array[minIndex] > array[j])
                    minIndex = j;
            //找到了最小的，进行交换
            swap(array, minIndex, i);
        }
        return array;
    }

2. 插入排序

打牌时，整理牌的算法。让未排序元素在已排序元素中找到合适位置并插入。

private void insertSort(int[] array) {
        //从第二个元素开始，因为第一个元素已经是有序的
        for (int i = 1; i < array.length; i++) {
            for (int j = i; j > 0 && array[j] < array[j - 1]; j--) {
                swap(array, j, j - 1);
            }
        }
    }
